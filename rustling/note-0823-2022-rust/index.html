Move semantics

fn main() {
    let vec0 = Vec::new();

    let mut vec1 = fill_vec(vec0); // add mut to make vec0 move to vec1

    println!("{} has length {} content `{:?}`", "vec1", vec1.len(), vec1);
  
    vec1.push(88);

    println!("{} has length {} content `{:?}`", "vec1", vec1.len(), vec1);
}

fn fill_vec(vec: Vec<i32>) -> Vec<i32> {
    let mut vec = vec;

    vec.push(22);
    vec.push(44);
    vec.push(66);

    vec
}
